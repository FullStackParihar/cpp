        cout << i << " ";

void swapValues(int& a, int& b) {

    b = temp;
}

bool isPalindrome(int n) {
    string str = to_string(n);
    string rev = string(str.rbegin(), str.rend());
    return str == rev;
}

#include <iostream>
#include <cmath>
using namespace std;

// Function to calculate factorial of a number
int factorial(int n) {
    int fact = 1;
    for (int i = 1; i <= n; i++) {
        fact *= i;
    }
    return fact;
}

// Function to check if a number is prime
bool isPrime(int n) {
    if (n <= 1) return false;
    for (int i = 2; i <= sqrt(n); i++) {
        if (n % i == 0) return false;
    }
    return true;
}

int main() {
    int num1, num2, num3, num4;
    cout << "Enter four numbers: ";
    cin >> num1 >> num2 >> num3 >> num4;

    // Calculate the sum of four numbers
    int sum = num1 + num2 + num3 + num4;

    if (sum > 100) {
        int a, b, c;
        cout << "Enter three more numbers: ";
        cin >> a >> b >> c;
        float avg = (a + b + c) / 3.0;
        if (avg > 50) {
            int fact = factorial(4);
            cout << "Factorial of 4: " << fact << endl;
            cout << (isPrime(fact) ? "Prime" : "Not prime") << endl;
        }
    } else {
        int prod = num1 * num2;
        cout << "Enter another number: ";
        int x;
        cin >> x;
        int result = prod + x;
        cout << "Result: " << result << endl;
        if (result % 2 != 0) {
            int square = result * result;
            cout << "Square of result: " << square << endl;
        }
    }

    return 0;
}

// Function to check if a number is a perfect square
bool isPerfectSquare(int n) {
    int sqRoot = sqrt(n);
    return (sqRoot * sqRoot == n);
}

int main() {
    int num1, num2, num3, num4, num5, num6;
    cout << "Enter six numbers: ";
    cin >> num1 >> num2 >> num3 >> num4 >> num5 >> num6;

    // Find the maximum of the six numbers
    int maxNum = max({num1, num2, num3, num4, num5, num6});

    if (maxNum > 75) {
        int a, b;
        cout << "Enter two more numbers: ";
        cin >> a >> b;
        int result = maxNum - a - b;
        if (result < 0) {
            result = abs(result);
            cout << "Factorial of absolute value: " << factorial(result) << endl;
        }
    }

    cout << "Enter three more numbers: ";
    cin >> num1 >> num2 >> num3;
    int sum = num1 + num2 + num3;
    if (sum == 0) {
        cout << (isPerfectSquare(sum) ? "Perfect square" : "Not a perfect square") << endl;
        if (isPerfectSquare(sum)) {
            int root = sqrt(sum);
            cout << "Square root: " << root << endl;
        }
    }

    return 0;
}
